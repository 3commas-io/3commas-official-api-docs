<StickyLayout.Main>

## Query Parameters

<ApiParam name='api_key' type='string' id="api_key">
  Filter by API-key.
</ApiParam>

<ApiParam name='page' type='integer' id="page">
  Page number for pagination. 
  
  Default: `1`
</ApiParam>

<ApiParam name='per_page' type='integer' id="page">
   Quantity of records on one page.

   Minimum: `1`, Maximum: `100`, Default: `100`
</ApiParam>
</StickyLayout.Main>

```json
GET /ver1/accounts
```
