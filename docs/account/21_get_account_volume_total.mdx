<Endpoint
  url="/ver1/accounts/:account_id/volume/total"
  method="GET"
  permissions={
    <>
      <Permission title="Permission" description="ACCOUNTS_READ" />
      <Permission title="Security" description="SIGNED" />

  }

Returns all-time trading volume for the specified account in BTC and USD.

<Separator />

<StickyLayout.Root>
  <StickyLayout.Main>
    ## Path Parameters

    <ApiParam name="account_id" type="integer" required>
      Unique 3Commas ID of this exchange account entity.
    </ApiParam>

    ## Response Parameters

    <ApiParam name="volume_btc" type="string" id="volume_btc">
      Total trading volume for the account since creation, in BTC.
    </ApiParam>

    <ApiParam name="volume_usd" type="string" id="volume_usd">
      Total trading volume for the account since creation, in USD.
    </ApiParam>
  </StickyLayout.Main>

  <StickyLayout.Sticky>
    ## Example Request

    ```http
    GET /ver1/accounts/123456/volume/total
    ```

      </ResponseTabs.TabItem>
    </ResponseTabs.Root>
  </StickyLayout.Sticky>
