<Endpoint
  url="/v2/smart_trades/:smart_trade_id/trades/:trade_id/close_by_market"
  method="POST"
  permissions={
    <>
      <Permission title="Permission" description="SMART_TRADES_WRITE" />
      <Permission title="Security" description="SIGNED" />
    </>
  }
/>

Closes a specific take profit trade within a SmartTrade at market price using its ID.

<Informer variant="warning">
The endpoint is available for Take Profit Trades for the statuses **to_process**, **order_placed**, and **trailing_activated**.
</Informer>

<StickyLayout.Root>
    <StickyLayout.Main>

        <ApiParam name='smart_trade_id' type='integer' id="smart_trade_id" required>
        Unique 3Commas ID for this SmartTrade entity.
        </ApiParam>
        <ApiParam name='trade_id' type='integer' id="trade_id" required>
        Unique 3Commas ID for Trade entity.
        </ApiParam>
    </StickyLayout.Main>
    <StickyLayout.Sticky>
    ## Example Request
        ```json
        POST /v2/smart_trades/12345678/trades/1047934729/close_by_market
        ```
    </StickyLayout.Sticky>
</StickyLayout.Root>
