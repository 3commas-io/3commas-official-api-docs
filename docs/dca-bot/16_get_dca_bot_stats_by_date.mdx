<Endpoint
  url="/ver1/bots/stats_by_date"
  method="GET"
  permissions={
    <>
      <Permission title="Permission" description="BOTS_READ" />
      <Permission title="Security" description="SIGNED" />
    </>
  }
/>

Returns profit data for the specified DCA Bot(s) over a defined number of days.

<Separator />

<StickyLayout.Root>
   <StickyLayout.Main>
    ## Query Parameters
    <ApiParam name='account_id' type='integer' id="account_id">
      Filters entities linked to the specified 3Commas exchange account.
    </ApiParam>
    <ApiParam name='bot_id' type='integer' id="bot_id">
      Filters entities linked to the specified ID DCA Bot entity.
    </ApiParam>
    <ApiParam name='date' type='string' id="date" required>
      Specific days for which profit data is requested for this bot or account.
    </ApiParam>
  </StickyLayout.Main>
  <StickyLayout.Sticky>
    ## Example Request

    <NoContentEndpoint method='GET' endpoint='/ver1/bots/stats_by_date?date=2024-11-06T04:00:00.000Z&bot_id=15577628' />

  </StickyLayout.Sticky>
</StickyLayout.Root>

<StickyLayout.Root>
   <StickyLayout.Main>
      ## Response Parameters
      <ApiParam name='profits_in_usd' type='object' id="profits_in_usd">
          Represents cumulative USD profits or losses from trading activities.
          <CollapsibleNestedParams>
            <ApiParam name='USDT' type="string" id='USDT'>
                Cumulative USD profit or loss from all trading activity.
            </ApiParam>    
        </CollapsibleNestedParams>
      </ApiParam>
      <ApiParam name='usd_profit' type='string' id="usd_profit">
      USD profit from deals for a specific day.
      </ApiParam>
      <ApiParam name='btc_profit' type='string' id="btc_profit">
      BTC profit from deals for a specific day.
      </ApiParam>
    </StickyLayout.Main>
