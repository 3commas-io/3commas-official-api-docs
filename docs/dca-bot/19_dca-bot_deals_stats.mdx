## Parameters response
<ApiParam name='active' type='integer' id="active">
The number of currently active deals.
</ApiParam>

<ApiParam name='completed' type='integer' id="completed">
The total number of completed deals.
</ApiParam>

<ApiParam name='panic_sold' type='integer' id="panic_sold">
The number of deals closed at market price.
</ApiParam>

<ApiParam name='from_currency_is_dollars' type='boolean' id="from_currency_is_dollars">
Shows whether the base currency for profit calculation is USD.
</ApiParam>

<CollapseAll>
<CollapseAllHeading> More attributes <CollapseAllButton /> </CollapseAllHeading>
<CollapsibleApiParam name='completed_deals_usd_profit' type='string' id="completed_deals_usd_profit">
The total profit from completed deals in USD.
</CollapsibleApiParam>

<CollapsibleApiParam name='completed_deals_btc_profit' type='string' id="completed_deals_btc_profit">
The total profit from completed deals in BTC.
</CollapsibleApiParam>

<CollapsibleApiParam name='funds_locked_in_active_deals' type='string' id="funds_locked_in_active_deals">
The total funds currently locked in active deals, in USD.
</CollapsibleApiParam>

<CollapsibleApiParam name='btc_funds_locked_in_active_deals' type='string' id="btc_funds_locked_in_active_deals">
The total funds currently locked in active deals, in BTC.
</CollapsibleApiParam>

<CollapsibleApiParam name='active_deals_usd_profit' type='string' id="active_deals_usd_profit">
The current profit or loss from active deals in USD.
</CollapsibleApiParam>

<CollapsibleApiParam name='active_deals_btc_profit' type='string' id="active_deals_btc_profit">
The current profit or loss from active deals in BTC.
</CollapsibleApiParam>
