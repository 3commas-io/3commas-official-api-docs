Returns a list of currently active trades.

<Informer variant="info">
  This endpoint returns only active (open) trades. For completed trades, use <code>GET /ver1/trades/history</code>.
</Informer>

<Separator />

<StickyLayout.Root>

## Query Parameters

<ApiParam name="account_id" type="integer" id="account_id">
  Optional filter for trades on a specific account. You can get IDs via <code>GET /ver1/accounts</code>.
</ApiParam>

<ApiParam name="pair" type="string" id="pair">
  Optional filter for trades on a specific trading pair (e.g., <code>USDT_BTC</code>).
</ApiParam>

<ApiParam name="order_by" type="string" id="order_by">
  Field to sort results by. Allowed values: <code>pair</code>, <code>created_at</code>. Default is <code>created_at</code>.
</ApiParam>

<ApiParam name="order_direction" type="string" id="order_direction">
  Sort direction: <code>asc</code> or <code>desc</code>. Default is <code>asc</code>.
</ApiParam>

<ApiParam name="limit" type="integer" id="limit">
  Number of records to return. Default is <code>10</code>.
</ApiParam>

<ApiParam name="offset" type="integer" id="offset">
  Number of records to skip for pagination. Default is <code>0</code>.
</ApiParam>

  </StickyLayout.Main>
