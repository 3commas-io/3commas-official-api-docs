Returns the list of Grid Bots. Use the body parameters to filter results.

   <StickyLayout.Main>

## Query Parameters

<ApiParam name="account_id" type="string" id="account_id">
  Unique 3Commas ID for this Grid Bot entity. Use a comma-separated list to
  specify multiple `account_id` values.
</ApiParam>

<ApiParam name="state" type="string" id="state">
  Return entities that match the specified state. Use `enabled` to get active
  bots and `disabled` to get inactive bots.
</ApiParam>

<ApiParam name="sort_by" type="string" id="sort_by">
  Specifies the field by which to sort the response.
  <CollapsibleNestedParams
    openName="Hide allowed values"
    collapsedName="Show allowed values"
  >
    * **current_profit** * **profit** * **bot_id** * **pair** * **created_at** *
    **updated_at** * **name**
  </CollapsibleNestedParams>
</ApiParam>

<ApiParam name='sort_direction' type='string' id="sort_direction">
   Sets the direction of order `ASC` or `DESC`.

   <ParameterRange defaultVal="DESC" />
</ApiParam>

<CollapseAll>
<CollapseAllHeading> More attributes <CollapseAllButton /> </CollapseAllHeading>

<CollapsibleApiParam name="limit" type="string" id="limit">
   Quantity of the records you want to get in response.

  <ParameterRange min={1} max={1000} defaultVal={10} />
</CollapsibleApiParam>

<CollapsibleApiParam name="offset" type="integer" id="offset">
  Used to specify the starting point for a set of records to return in a
  paginated list.
  <ParameterRange defaultVal={0} />
</CollapsibleApiParam>

<CollapsibleApiParam name="from" type="string" id="from">
  Parameter for a filter by created date.
</CollapsibleApiParam>

<CollapsibleApiParam name="base" type="string" id="base">
  Return entities that match the base currency of the trading pair (e.g., `BTC`
  for BTC/USDT).
</CollapsibleApiParam>

<CollapsibleApiParam name="quote" type="string" id="quote">
   Return entities that match the quote currency of the trading pair (e.g., `USDT` for BTC/USDT).
</CollapsibleApiParam>
</CollapseAll>
